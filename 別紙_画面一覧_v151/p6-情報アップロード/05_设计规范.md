    @Async
    @Transactional
    public UploadFileResponse uploadFile(UploadFileRequest request, String partnerId) {

        UploadHistory uploadHistory = UploadHistory.builder()
                .uploadId(uploadId)

                .memo(request.getMemo())
                .uploadedBy(partnerId)
                .build();

        fileProcessingService.processFileAsync(uploadId, request.getFile(), partnerId);
        
        return UploadFileResponse.builder()
                .success(true)

            UploadHistorySpecification.hasPartnerId(partnerId));

                                .totalPages(page.getTotalPages())
                                .build())

    }
