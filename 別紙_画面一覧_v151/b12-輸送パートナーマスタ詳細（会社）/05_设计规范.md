    @Transactional(readOnly = true)

        PartnerCompany company = partnerCompanyRepository.findById(companyId)

                        .accountStatus(user.getAccountStatus().getValue())

                .success(true)
                .data(PartnerCompanyData.builder()
                        .companyId(company.getCompanyId())
                        .companyName(company.getCompanyName())
                        .companyCodes(company.getCompanyCodes())
                        .companyPhone(company.getCompanyPhone())
                        .systemUse(company.isSystemUse())
