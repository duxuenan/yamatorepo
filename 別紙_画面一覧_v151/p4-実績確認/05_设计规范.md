@RestController
@RequestMapping("/api/v1/partners/results")
@RequiredArgsConstructor
public class PartnerResultController {

    private final PartnerResultService partnerResultService;

    @GetMapping
    public ResponseEntity<GetResultsResponse> getResults(
            @RequestParam(required = false) String yearMonth,
            @RequestParam(required = false) String baseId,
            @RequestParam(required = false) String status,
            @RequestParam(defaultValue = "1") int page,
            @RequestParam(defaultValue = "20") int pageSize,
            @AuthenticationPrincipal User user) {
        GetResultsResponse response = partnerResultService.getResults(
            yearMonth, baseId, status, page, pageSize, user.getPartnerId());
        return ResponseEntity.ok(response);
    }

    @GetMapping("/{resultId}/daily-costs")
    public ResponseEntity<GetDailyCostsResponse> getDailyCosts(
            @PathVariable String resultId,
            @AuthenticationPrincipal User user) {
        GetDailyCostsResponse response = partnerResultService.getDailyCosts(
            resultId, user.getPartnerId());
        return ResponseEntity.ok(response);
    }

    @PostMapping("/{resultId}/confirm")
    public ResponseEntity<ConfirmResultResponse> confirmResult(
            @PathVariable String resultId,
            @RequestBody ConfirmResultRequest request,
            @AuthenticationPrincipal User user) {
        ConfirmResultResponse response = partnerResultService.confirmResult(
            resultId, request, user.getPartnerId(), user.getName());
        return ResponseEntity.ok(response);
    }

    @PutMapping("/{resultId}/daily-costs/{costId}")
    public ResponseEntity<ChangeDailyCostResponse> changeDailyCost(
            @PathVariable String resultId,
            @PathVariable String costId,
            @RequestBody ChangeDailyCostRequest request,
            @AuthenticationPrincipal User user) {
        ChangeDailyCostResponse response = partnerResultService.changeDailyCost(
            resultId, costId, request, user.getPartnerId());
        return ResponseEntity.ok(response);
    }
}
